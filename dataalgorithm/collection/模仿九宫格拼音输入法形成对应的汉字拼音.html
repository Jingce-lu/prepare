var letterCombinations = num => {
  let str = String(num);
  if(str.includes("1") || str.includes("0")) {
    return new Error("不能包含0或1");
  }
  if(str.length === 0){
    return [];
  }

  let results = [];
  let first = str[0];
  let arr = {
    '2': 'abc',
    '3': 'def',
    '4': 'ghi',
    '5': 'jkl',
    '6': 'mno',
    '7': 'pqrs',
    '8': 'tuv',
    '9': 'wxyz'
  }
  // 当长度为1时，直接返回此数字代表的字符数组
  if (str.length == 1) { 
    for (let i = 0; i < arr[first].length; i++) {
      results.push(arr[first][i])
    }
  }

  for (let i = 0; i < arr[first].length; i++) {
    let result = letterCombinations(str.slice(1))
    for (let j = 0; j < result.length; j++) {
      results.push(arr[first][i] + result[j])
    }
  }
  return results
}
