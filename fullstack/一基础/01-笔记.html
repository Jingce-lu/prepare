    window.onload = function(){
  var oBtn = document.getElementById('btn1');
  var oF1 = document.getElementById('f1');

  oBtn.onclick = function(){
    var formData = new FormData();
    formData.set('f1', oF1)

    var xhr = new XMLHttpRequest();

    xhr.upload.onload = function() {
      console.log('上传完成')
    }

    xhr.upload.onprogress = function(ev) {
      console.log(ev.loaded, ev.total) // 上传进度
    }

    xhr.open('post', '1.php', true);
    xhr.send(formData)  

    xhr.onreadystatechange = function () {
      if(xhr.readyState === 4) {
        if(xhr.status >= 200 && xhr.status < 300 || xhr.status == 304) {
          console.log(xhr.responseText)
        } else {
          console.log('出错了')
        }
      }
    }
  }
}

// 服务端
const http = require('http')
const url = require('url')

let server = http.createServer(function(req, res){
  let { pathname, query } = url.parse(req.url, true);
  let { a, b, callback } = query;

  res.write(`${callback}(${a+b})`);
  res.end()
})

server.listen(8989)

# webwork

webwork -- 浏览器实现的多进程

web端：

  • 主进程 ui进程， 唯一只有一个
  • 子进程(工作进程) 看不见；只能完成计算、数据请求操作

优点：

  1. 充分利用资源(多个进程同时工作)
  2. 防止主进程卡主
<!DOCTYPE html>
<html lang="en">
<head>
	<meta charset="UTF-8">
	<title>webworker</title>
	<script>
	window.onload = function(){
		let oTxt1 = document.getElementById('txt1')
		let oTxt2 = document.getElementById('txt2')
		let oBtn = document.getElementById('btn1')

		oBtn.onclick = function(){
			let n1 = parseInt(oTxt1.value);
			let n2 = parseInt(oTxt2.value);

			// 1.创建子进程
			let work = new Worker('w1.js');
			// 2.发送
			work.postMessage({n1, n2})

			// 6.接收结果
			work.onmessage = function(ev){
				console.log(ev.data)
			}
		}
	}
	</script>
</head>

# SVG

svg图形：

1. 样式操作     跟HTML一样 
2. 事件操作     跟HTML一样  
3. 属性操作     有点区别
            HTML                    SVG
  设置    .xxx = xxx            .setAttribute
          .setAttribute
  设置    .xxx           .getAttribute
          .getAttribute
4. DOM操作
  获取      完全一样
  创建      HTML： createElement
            SVG:  createElementNS('http://www.w3.org/2000/svg', 'line')
  插入      完全一样
5.事件、修改、大部分操作跟HTML一样
  不同： attribute、 createElementNS('网址'， '标签')
